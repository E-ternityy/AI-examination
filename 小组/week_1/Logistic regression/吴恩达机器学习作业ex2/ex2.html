# sigmoid函数
def sigmoid(z):
    """
    sigmoid函数
    参数:
        z(np.array): 输入

    返回:
        array: 结果
    """
    return 1 / (1 + np.exp(-z))

def cost(theta, X, y):
    """
    代价函数

    参数:
        theta (np.array): 模型参数
        X (np.array): 特征
        y (np.array): 真实值

    返回:
        float: 代价值
    """
    theta = np.matrix(theta)
    X = np.matrix(X)
    y = np.matrix(y)
    first = np.multiply(-y, np.log(sigmoid(X @ theta)))
    second = np.multiply((1 - y), np.log(1 - sigmoid(X @ theta)))
    return np.sum(first - second) / (len(X))

def gradientDescent(X, y, theta, alpha, times):
    """
    梯度下降算法优化参数

    参数:
        X (np.array): 特征
        y (np.array): 真实值
        theta (np.array): 参数
        alpha (float): 学习率
        times (int): 迭代次数

    返回:
        np.array: 模型参数
    """
    m = len(X)
    for i in range(times): # 迭代
        A = sigmoid(X @ theta)
        theta = theta - (alpha / m) * X.T @ (A - y)

    fcost = cost(theta, X, y)
    return fcost, theta

def predict(X, theta):
    """
    使用训练好的模型进行预测

    参数:
        X (np.array): 特征
        theta (np.array): 模型参数

    返回:
        list: 预测结果，1表示录取，0表示不录取
    """
    prob = sigmoid(X @ theta)
    return [1 if x >= 0.5 else 0 for x in prob]
